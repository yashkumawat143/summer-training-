5....intersection of array.
answer :


#include<iostream>
#include<set>
using namespace std;
int main(){
    set<int> s;
    int a,b;
    cout<<"Enter the size of first array : ";
    cin>>a;
    int arr[a];
    for(int i=0;i<a;i++){
        cout<<"Enter the elements : ";
        cin>>arr[i];
    }
    cout<<"Enter the size of second array : ";
    cin>>b;
    int arr1[b];
    for(int i=0;i<b;i++){
        cout<<"Enter the elements : ";
        cin>>arr1[i];

    }
    for(int i=0;i<a;i++){
        s.insert(arr[i]);
    }
    for(int i=0;i<b;i++){
        if(s.find(arr1[i])!=s.end()){
            cout<<arr1[i]<<" ";
            s.erase(arr1[i]);

        }
    }
}
