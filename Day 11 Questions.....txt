2....character frequency.
answer :

#include<iostream>
#include<string>
using namespace std;
int main(){
    string s;
    cout<<"Enter the element : ";
    cin>>s;
    for(int i=0;i<s.length();i++){
        int count=1;
        for(int j=i+1;j<s.length();j++){
            if(s[i]==s[j]){
                count++;
            }
        }
        cout<<s[i]<<"="<<count<<endl;
    }
}



3....most frequent element.
answer :


#include<iostream>
#include<string>
using namespace std;
int main(){
    string s;
    cout<<"Enter the element : ";
    cin>>s;
    int max=0;
    for(int i=0;i<s.length();i++){
        int count=1;
        
        for(int j=i+1;j<s.length();j++){
            if(s[i]==s[j]){
                count++;
                
            }
        }
        if(count>max){
            max=count;
        }
        cout<<s[i]<<"="<<count<<endl;
    }
    cout<<"The maximum element = "<<max;
    

}

5....word frequency.
answer :

#include<iostream>
#include<string>
#include<map>
#include<sstream>
using namespace std;
int main(){
    string s;
    cout<<"Enter the words : ";
    getline(cin,s);
    map<string,int>fre;
    stringstream ss(s);
    string word;
    while(ss>>word){
        fre[word]++;
    }
    for(auto x:fre){
        cout<<x.first<<"="<<x.second<<endl;
    }

}
