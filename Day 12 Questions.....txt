        if(amount>=b){

            cout<<"Amount has been successfully withdraw";

};

5....car + static.
answer :


#include <iostream>
using namespace std;

class Car {
private:
    static int count;   

public:
    Car() {
        count++;       
    }

    static void showCount() {
        cout << "Total Car objects created: " << count << endl;
    }
};


int Car::count = 0;

int main() {
    Car c1;
    Car c2;
    Car c3;

    Car::showCount();

    return 0;
}
