3....Array sum using pointer.
answer :


#include<iostream>
using namespace std;
int main(){
    int length;
    int *ptr;
    cout<<"Enter the size : ";
    cin>>length;
    int a[length];
    for(int i=0;i<length;i++){
        cout<<"Enter the number : ";
        ptr=&a[i];
        cin>>*ptr;
    }
    int sum=0;
    for(int i=0;i<length;i++){
        ptr=&a[i];
        sum+=*ptr;
        
    }
    cout<<sum;

}



4....largest element using pointer.
answer :

#include<iostream>
using namespace std;
int main(){
    int length;
    cout<<"Enter the size : ";
    cin>>length;
    int a[length];
    int *ptr;
    for(int i=0;i<length;i++){
        cout<<"Enter the number : ";
        ptr=&a[i];
        cin>>*ptr;
    }
    int max=a[0];
    for(int i=0;i<length;i++){
        ptr=&a[i];
        if(max<*ptr){
            max=*ptr;
        }
    }
    cout<<"Largest value : "<<max;    


}

6....reverse array using pointer.
answer :

#include<iostream>
using namespace std;
int main(){
    int length;
    int *ptr;
    cout<<"Enter the size : ";
    cin>>length;
    int a[length];
    for(int i=0;i<length;i++){
        cout<<"Enter the number : ";
        ptr=&a[i];
        cin>>*ptr;

    }
    for(int i=length-1;i>=0;i--){
        ptr=&a[i];
        cout<<*ptr<<" ";
        
    }



}
