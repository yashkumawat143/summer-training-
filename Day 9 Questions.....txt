4....remove duplicate.
answer :

#include<iostream>
#include<vector>
using namespace std;
int main(){
    vector<int> v;
    int n;
    cout<<"Enter the size : ";
    cin>>n;
    for(int i=0;i<n;i++){
        int x;
        cout<<"Enter the number : ";
        cin>>x;
        v.push_back(x);
    }
    for(int i=0;i<v.size();i++){
        for(int j=i+1;j<v.size();j++){
            if(v[i]==v[j]){
                v.erase(v.begin()+j);
                j--;
            }
        }
    }
    for(int i=0;i<v.size();i++){
        cout<<v[i]<<" ";
    }

}

6....rotate vector.
answer :

#include<iostream>
#include<vector>
using namespace std;
int main(){
    vector<int> v;
    int n;
    cout<<"Enter the size : ";
    cin>>n;
    for(int i=0;i<n;i++){
        int x;
        cout<<"Enter the number : ";
        cin>>x;
        v.push_back(x);
    }
    vector<int> ans(n);
    int k;
    cout<<"Enter the position of roatation : ";
    cin>>k;
    for(int i=0;i<n;i++){
        ans[(k+i)%n]=v[i];

    }
    for(int i=0;i<n;i++){
        cout<<ans[i]<<" ";
    }
}
